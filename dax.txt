DEFINE
    MEASURE '_Medidas Base y de Rentabilidad'[Revenue Total] = SUM(orders_clean[monto_total])
    MEASURE '_Medidas Base y de Rentabilidad'[Costo Produccion Total] = SUM(orders_clean[Costo Produccion Pedido])
    MEASURE '_Medidas Base y de Rentabilidad'[Inversion Marketing] = SUM(marketing_clean[gasto])
    MEASURE '_Medidas Base y de Rentabilidad'[Profit Total] = IF(
    HASONEVALUE(orders_clean[id_pedido]) || ISFILTERED(orders_clean[nombre_producto]), 
    [Revenue Total] - [Costo Produccion Total], 
    [Revenue Total] - ([Costo Produccion Total] + [Inversion Marketing])
)
    MEASURE '_Medidas Base y de Rentabilidad'[Profit Margin %] = DIVIDE([Profit Total], [Revenue Total], 0)
    MEASURE '_Medidas Base y de Rentabilidad'[Ganancia Bruta] = [Revenue Total] - [Costo Produccion Total]
    MEASURE '_Medidas de Ventas'[Ticket Promedio] = AVERAGE(orders_clean[monto_total])
    MEASURE '_Medidas de Ventas'[Cantidad Promedio] = AVERAGE(orders_clean[cantidad])
    MEASURE '_Medidas de Ventas'[Cantidad Total Vendida] = SUM(orders_clean[cantidad])
    MEASURE '_Medidas de Inteligencia de Tiempo'[Revenue YTD] = VAR MaxDateWithData = CALCULATE(MAX(orders_clean[Fecha Pedido]), ALL(orders_clean))
RETURN
IF (
    MAX('Calendario'[Date]) <= MaxDateWithData,
    TOTALYTD([Revenue Total], 'Calendario'[Date]),
    BLANK()
)
    MEASURE '_Medidas de Inteligencia de Tiempo'[Profit YTD] = VAR MaxDateWithData = CALCULATE(MAX(orders_clean[Fecha Pedido]), ALL(orders_clean))
RETURN
IF (
    MAX('Calendario'[Date]) <= MaxDateWithData,
    TOTALYTD([Profit Total], 'Calendario'[Date]),
    BLANK()
)
    MEASURE '_Medidas de Inteligencia de Tiempo'[Revenue LY] = CALCULATE([Revenue Total], SAMEPERIODLASTYEAR('Calendario'[Date]))
    MEASURE '_Medidas de Inteligencia de Tiempo'[Revenue YoY Crecimiento %] = DIVIDE([Revenue Total] - [Revenue LY], [Revenue LY], 0)
    MEASURE '_Medidas_extra'[Margen Neto %] = DIVIDE([Profit Total], [Revenue Total])
    MEASURE '_Medidas_extra'[Profit por Unidad] = DIVIDE([Profit Total], [Cantidad Total Vendida])
    MEASURE '_Medidas_extra'[Revenue por Pedido] = AVERAGE(orders_clean[monto_total])
    MEASURE '_Medidas_extra'[Profit por $ Mkt] = DIVIDE([Profit Total], [Inversion Marketing])
    MEASURE '_Medidas_extra'[Profit % Pais] = DIVIDE([Profit Total], CALCULATE([Profit Total], ALL(Dim_Pais)))
    MEASURE '_Medidas_extra'[% Acumulado] = VAR IngresosTotales = CALCULATE(SUM(Pareto_Productos[Revenue_Producto]), ALLSELECTED(Pareto_Productos))
VAR IngresosActuales = SUM(Pareto_Productos[Revenue_Producto])
VAR Acumulado = 
    CALCULATE(
        SUM(Pareto_Productos[Revenue_Producto]),
        FILTER(
            ALLSELECTED(Pareto_Productos),
            Pareto_Productos[Revenue_Producto] >= IngresosActuales
        )
    )
RETURN
DIVIDE(Acumulado, IngresosTotales)
    MEASURE '_Medidas_extra'[Monto_Waterfall] = VAR Seleccion = SELECTEDVALUE(Conceptos_Waterfall[Concepto])
RETURN
SWITCH( Seleccion,
    "1. Revenue Total", [Revenue Total],
    "2. Costo Produccion Total", -[Costo Produccion Total], -- El signo menos crea la caída
    "3. Inversión Marketing", -[Inversion Marketing], -- El signo menos crea la caída
    BLANK()
)
    MEASURE '_Medidas_extra'[ROI por Canal %] = DIVIDE(
    [Profit Total], 
    [Inversion Marketing], 
    0
)

EVALUATE
    SUMMARIZECOLUMNS(
        "Revenue Total", [Revenue Total],
        "Costo Produccion Total", [Costo Produccion Total],
        "Inversion Marketing", [Inversion Marketing],
        "Profit Total", [Profit Total],
        "Profit Margin %", [Profit Margin %],
        "Ganancia Bruta", [Ganancia Bruta],
        "Ticket Promedio", [Ticket Promedio],
        "Cantidad Promedio", [Cantidad Promedio],
        "Cantidad Total Vendida", [Cantidad Total Vendida],
        "Revenue YTD", [Revenue YTD],
        "Profit YTD", [Profit YTD],
        "Revenue LY", [Revenue LY],
        "Revenue YoY Crecimiento %", [Revenue YoY Crecimiento %],
        "Margen Neto %", [Margen Neto %],
        "Profit por Unidad", [Profit por Unidad],
        "Revenue por Pedido", [Revenue por Pedido],
        "Profit por $ Mkt", [Profit por $ Mkt],
        "Profit % Pais", [Profit % Pais],
        "% Acumulado", [% Acumulado],
        "Monto_Waterfall", [Monto_Waterfall],
        "ROI por Canal %", [ROI por Canal %]
    )
